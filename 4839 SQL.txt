$SELECT V_PO_header.PURCHASE_ORDER AS "PO No", V_PO_header.VENDOR AS "Vendor", v_po_vendor_buy.name_vendor as "Name", V_PO_LINES.RECORD_NO AS "Line No", V_PO_LINES.PART AS "Part No", V_PO_LINES.date_due_line AS "Date Due", v_po_lines.job + '- ' + v_po_lines.suffix as "Job No", v_job_header.Project,v_po_lines.Extension,v_po_lines.qty_order - v_po_lines.qty_received - qty_recd_not_insp as "Qty Open", v_po_lines.qty_order as "Order Qty", v_po_lines.qty_received as "Rec Qty", v_po_lines.qty_recd_not_insp as "Rec Not Insp" FROM V_PO_lines Left Join V_PO_header On v_po_lines.purchase_order = V_PO_header.purchase_order Left Join v_po_vendor_buy On v_po_vendor_buy.purchase_order = V_po_header.purchase_order Left Join v_job_header On v_po_lines.job = v_job_header.job And v_po_lines.suffix = v_job_header.suffix ORDER BY "PO No"
